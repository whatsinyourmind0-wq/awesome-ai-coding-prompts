**Problem**: Write a thread-safe in-memory cache with TTL in Go.
**Claude**:
```go
package cache

import (
	"sync"
	"time"
)

type item struct {
	value      interface{}
	expiration int64
}

type TTLMap struct {
	m map[string]*item
	l sync.RWMutex
}

func NewTTLMap() *TTLMap {
	return &TTLMap{m: make(map[string]*item)}
}

func (m *TTLMap) Put(k string, v interface{}, ttl time.Duration) {
	m.l.Lock()
	defer m.l.Unlock()
	m.m[k] = &item{
		value:      v,
		expiration: time.Now().Add(ttl).UnixNano(),
	}
}

func (m *TTLMap) Get(k string) (interface{}, bool) {
	m.l.RLock()
	defer m.l.RUnlock()
	if it, ok := m.m[k]; ok {
		if time.Now().UnixNano() > it.expiration {
			return nil, false // Expired
		}
		return it.value, true
	}
	return nil, false
}
```
