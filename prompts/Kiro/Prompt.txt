## EXAMPLE SNIPPET (Svelte 5)
```svelte
<script lang="ts">
  let x = $state(0);
  let y = $state(0);
  
  function handleMouse(e: MouseEvent) {
    x = e.clientX;
    y = e.clientY;
  }
</script>

<svelte:window onmousemove={handleMouse} />

<div 
  class="pointer" 
  style="transform: translate({x}px, {y}px)"
></div>

<style>
  .pointer {
    position: absolute;
    width: 20px;
    height: 20px;
    background-color: var(--accent-color, #ff3e00);
    border-radius: 50%;
    transition: transform 0.1s cubic-bezier(0.25, 0.1, 0.25, 1);
  }
</style>
```
